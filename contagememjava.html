import java.util.Scanner; public class Main { public static void main(String[] args) { Scanner sc = new Scanner(System.in); // Lê o valor de N int n = sc.nextInt(); // Lê a lista de números de entrada int[] nums = new int[n]; for (int i = 0; i < n; i++) { nums[i] = sc.nextInt(); } // Calcula a quantidade de números inteiros entre o primeiro dado e N int count = 0; for (int i = 0; i < n; i++) { if (nums[i] >= nums[0] && nums[i] <= n) { count++; } } // Imprime o resultado System.out.println(count); } }
